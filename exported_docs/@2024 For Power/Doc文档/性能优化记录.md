SELECT w.*,
       z.mainType,
       z.NAME  AS               customerName,
       z.phone as               customerPhone,
       z.contactCertificateType certificateType,
       z.contactCertificateCard certificateCard,
       z.connectId              connectId
FROM (SELECT a.F_Id              AS id,
             a.F_ProjectId       AS projectId,
             a.F_RegionId        AS regionId,
             a.F_BuildingId      AS buildingId,
             a.F_UnitId          AS unitId,
             a.F_FloorId         AS floorId,
             a.F_ResourcesId     AS resourcesId,
             a.F_RoomFullName    AS roomFullName,
             a.F_ChargeStartTime AS chargeStartTime,
             a.F_CustomerId      AS customerId,
             a.F_OrganizeId      AS organizeId,
             a.F_OwnerType       AS ownerType,
             a.F_UseStatus       AS useStatus,
             a.F_PartnershipTime AS partnershipTime,
             a.F_LeaseStartTime  AS leaseStartTime,
             a.F_LeaseEndTime    AS leaseEndTime,
             a.F_MoveDate        AS moveDate,
             a.F_TransferDate    as transferDate,
             a.F_DeleteMark      AS deleteMark,
             '1'                 AS resourceMode,
             a.F_SourceType      AS sourceType,
             a.F_CreatorTime        creatorTime
      FROM room_owner a
      UNION
      SELECT b.F_Id              AS id,
             b.F_ProjectId       AS projectId,
             b.F_RegionId        AS regionId,
             b.F_BuildingId      AS buildingId,
             b.F_UnitId          AS unitId,
             b.F_FloorId         AS floorId,
             b.F_CarPostId       AS resourcesId,
             b.F_CarPostFullName AS roomFullName,
             b.F_ChargeStartTime AS chargeStartTime,
             b.F_CustomerId      AS customerId,
             b.F_OrganizeId      AS organizeId,
             b.F_OwnerType       AS ownerType,
             b.F_UseStatus       AS useStatus,
             b.F_PartnershipTime AS partnershipTime,
             b.F_LeaseStartTime  AS leaseStartTime,
             b.F_LeaseEndTime    AS leaseEndTime,
             b.F_MoveDate        AS moveDate,
             b.F_TransferDate    as transferDate,
             b.F_DeleteMark      AS deleteMark,
             '2'                 AS resourceMode,
             b.F_SourceType      AS sourceType,
             b.F_CreatorTime        creatorTime
      FROM car_owner b) w
         LEFT JOIN (SELECT c.F_Id                                                               AS customerId,
                           c.F_MainType                                                         AS mainType,
                           c.F_OrganizeId                                                       AS organizeId,
                           p.F_ContactName                                                      AS contactName,
                           e.F_CompanyName                                                      AS companyName,
                           IF(c.F_MainType = 1, p.F_ContactName, e.F_CompanyName)               AS NAME,
                           IF(c.F_MainType = 1, p.F_ContactTel, e.F_EnterprisePhone)            AS phone,
                           IF(c.F_MainType = 1, p.F_ContactCertificateType, e.F_DocumentType)   AS contactCertificateType,
                           IF(c.F_MainType = 1, p.F_ContactCertificateCard, e.F_DocumentNumber) AS contactCertificateCard,
                           IF(c.F_MainType = 1, p.F_CustomerContactId, e.F_CustomerCompany)     AS connectId
                    FROM customer c
                             LEFT JOIN personal p
                                       ON c.F_Id = p.F_CustomerId AND p.F_DeleteMark = 0 AND p.F_EnabledMark = 1
                             LEFT JOIN enterprise e ON c.F_Id = e.F_CustomerId
                             LEFT JOIN customer_contacts r
                                       ON r.F_CustomerId = c.F_Id AND r.F_ContactType = 1 and r.F_DeleteMark = 0
                    WHERE c.F_DeleteMark = 0
                      AND c.F_EnabledMark = 1
                    GROUP BY c.F_Id) z ON w.customerId = z.customerId
WHERE w.deleteMark = 0
  AND w.organizeId = '29db9ccaa9b24cb796f1f1f1ae4df305'
  AND w.useStatus in (0)
ORDER BY w.customerId DESC, w.creatorTime DESC
LIMIT 20;
