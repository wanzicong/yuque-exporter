    public Map<Long, PmsResourcesType> getResourcesTypeMap(Collection<Long> resourceIds) {
        Map<Long, PmsResourcesType> resourceTypeMap = new HashMap<>();
        if (ToolUtil.isEmpty(resourceIds)) {
            return resourceTypeMap;
        }
        // 产权业态
        List<PmsResourcesByIdsResp> resoureList = getResListByIds(resourceIds);
        if (ToolUtil.isNotEmpty(resoureList)) {
            Set<Long> resTypeIds = resoureList.stream()
                    .filter(x -> ToolUtil.isNotEmpty(x.getResourcesType()))
                    .map(x -> Long.parseLong(x.getResourcesType()))
                    .collect(Collectors.toSet());
            ActionResult<Map<Long, PmsResourcesType>> resourcesTypeActionResult = pmsResourcesTypeApi.getResourcesTypeMap(new IdsReq() {{
                setIds(resTypeIds);
            }});
            if (resourcesTypeActionResult != null && ToolUtil.isNotEmpty(resourcesTypeActionResult.getData())) {
                Map<Long, PmsResourcesType> typeMap = resourcesTypeActionResult.getData();
                for (PmsResourcesByIdsResp resource : resoureList) {
                    if (ToolUtil.isNotEmpty(resource.getResourcesType())) {
                        PmsResourcesType resourcesType = typeMap.get(Long.parseLong(resource.getResourcesType()));
                        if (ToolUtil.isNotEmpty(resourcesType)) {
                            resourceTypeMap.put(Long.parseLong(resource.getId()), resourcesType);
                        }
                    }
                }
            }
        }
        return resourceTypeMap;
    }

    private void fillPropertyBillInfo(List<OrderBillResp> list) {
        if (ToolUtil.isEmpty(list)) {
            return;
        }
        Set<Long> billIds = list.stream().filter(x -> ObjectUtil.equal(x.getOrderType(), FinanceEnums.OrderTypeEnum.ESTATE.getValue()))
                .map(OrderBillResp::getId).collect(Collectors.toSet());
        if (ToolUtil.isEmpty(billIds)) {
            return;
        }
        Map<Long, OrderBillPropertyEntity> propertyMap = orderBillPropertyService.getPropertyEntityMap(billIds);
        for (OrderBillResp resp : list) {
            if (resp == null) {
                continue;
            }
            OrderBillPropertyEntity billProperty = propertyMap.get(resp.getId());
            if (billProperty == null) {
                continue;
            }
            resp.setDosage(billProperty.getDosage());
            resp.setRangePrice(billProperty.getRangePrice());
            if (ToolUtil.isEmpty(billProperty.getDetailJson())) {
                continue;
            }
            List<TaskDetailItemBO> detailList = null;
            try {
                detailList = JSON.parseArray(billProperty.getDetailJson(), TaskDetailItemBO.class);
            } catch (Exception e) {
                e.printStackTrace();
            }
            if (ToolUtil.isEmpty(detailList)) {
                continue;
            }
            resp.setMeterName(detailList.stream().filter(x -> ToolUtil.isNotEmpty(x.getSerialNumber()))
                    .map(TaskDetailItemBO::getSerialNumber).distinct().collect(Collectors.joining(",")));

            if (ToolUtil.isEmpty(resp.getDosage())) {
                BigDecimal consumeValue = detailList.stream().filter(x -> ToolUtil.isNotEmpty(x.getConsumeValue()))
                        .map(TaskDetailItemBO::getConsumeValue).reduce(BigDecimal.ZERO, BigDecimal::add);
                resp.setDosage(consumeValue != null ? consumeValue.toPlainString() : "");
            }

            if (ToolUtil.isEmpty(resp.getRangePrice())) {
                String price = detailList.stream().filter(x -> ToolUtil.isNotEmpty(x.getPrice()) && ToolUtil.isNotEmpty(x.getUnit()))
                        .map(x -> x.getPrice().toPlainString() + x.getUnit()).distinct().collect(Collectors.joining(","));
                resp.setRangePrice(ToolUtil.isNotEmpty(price) ? price : "");
            }
        }

    }

    private void formatResult(List<OrderBillResp> list, Boolean needReceipt) {
        if (ToolUtil.isEmpty(list)) {
            return;
        }
        List<String> orgIds = list.stream().filter(x -> ToolUtil.isNotEmpty(x.getOrganizeId()))
                .map(OrderBillResp::getOrganizeId).distinct().collect(Collectors.toList());
        Map<String, String> orgMap = organizeApi.getOrgMap(orgIds);
        list.forEach(x -> {
            if (ToolUtil.isNotEmpty(x.getOrganizeId())) {
                x.setOrganizeName(orgMap.get(x.getOrganizeId()));
            }
        });
        // 填充账单详情
        orderBillDetailService.fillBillDetail(list);
        // 填充已开收据金额
        if (needReceipt != null && needReceipt) {
            Set<Long> billIds = list.stream().map(OrderBillResp::getId).collect(Collectors.toSet());
            Map<Long, List<FinanceReceiptContractEntity>> receiptGroupMap = new HashMap<>();
            if (ToolUtil.isNotEmpty(billIds)) {
                List<FinanceReceiptContractEntity> allReceiptList = financeReceiptContractService.getBaseMapper().getListByBillIds(billIds);
                if (ToolUtil.isNotEmpty(allReceiptList)) {
                    receiptGroupMap = allReceiptList.stream().collect(Collectors.groupingBy(FinanceReceiptContractEntity::getBillId));
                }
            }

            Map<Long, List<FinanceReceiptContractEntity>> finalReceiptGroupMap = receiptGroupMap;
            list.forEach(x -> {
                BigDecimal received = x.getAmountActuallyReceived() == null ? BigDecimal.ZERO : x.getAmountActuallyReceived();
                List<FinanceReceiptContractEntity> receiptList = finalReceiptGroupMap.get(x.getId());
                BigDecimal receiptAmount = null;
                if (ToolUtil.isNotEmpty(receiptList)) {
                    receiptAmount = receiptList.stream().map(FinanceReceiptContractEntity::getAssociatedAmount).reduce(BigDecimal.ZERO, BigDecimal::add);
                    x.setReceiptAmount(receiptAmount);
                } else {
                    x.setReceiptAmount(BigDecimal.ZERO);
                }
                BigDecimal receipt = receiptAmount == null ? BigDecimal.ZERO : receiptAmount;
                x.setAssociatedAmount(received.subtract(receipt));
            });
        }
    }

    public Boolean delOrderBill(List<OrderBillEntity> list) {
        if (ToolUtil.isEmpty(list)) {
            return false;
        }
        boolean flag = this.removeBatchByIds(list, list.size());
        if (flag) {
            Set<Long> billIds = list.stream().map(OrderBillEntity::getId).collect(Collectors.toSet());
            // 删除账单详情

            billReceivableDetailService.removeImportData(billIds);

            orderBillPropertyService.dealOrderMainCancelData(orderMainIds);
            // 更新账单状态

        }
        return flag;
    }

            item.setExpenseName(x.getExpenseName());
            if (x.getLeaseStartTime() != null && x.getLeaseEndTime() != null) {
                String s = String.format("%s-%s", DateUtil.formatDate(x.getLeaseStartTime()), DateUtil.formatDate(x.getLeaseEndTime()));
                item.setLeaseDuration(s);
            }
            item.setProjectName(x.getProjectName());
            item.setBulidingName(x.getBuildingName());
            item.setRegionName(x.getRegionName());
            item.setUnitName(x.getUnitName());
            item.setOrganizeName(x.getOrganizeName());
            item.setFloorName(x.getFloorName());
            item.setResourcesName(x.getResourcesName());
            item.setLesseeInfoName(x.getLesseeInfoName());
            item.setConnectTel(x.getConnectTel());
            item.setActualAmount(UtilHelps.formatAmount(x.getActualAmount(), x.getRetainType()));
            item.setAmountActuallyReceived(UtilHelps.formatAmount(x.getAmountActuallyReceived(), x.getRetainType()));
            item.setAmountToBeReceived(x.getAmountToBeReceived());
            if (x.getPayEndTime() != null) {
                item.setPayEndTime(DateUtil.formatDate(x.getPayEndTime()));
            }
            item.setPayState(x.getPayStateDesc());
            item.setMeterFullName(x.getMeterFullName());

            list.add(item);
        }

                .businessTypeEnum(Enums.FileBusinessTypeEnum.FINANCE_ORDER_BILL)

    public Boolean refreshOrderBillPayState() {
        log.info("开始更新已逾期账单状态");
        // 查询已逾期的账单
        List<OrderBillEntity> list = this.list(Wrappers.<OrderBillEntity>lambdaQuery()
                .eq(OrderBillEntity::getPayState, FinanceEnums.FinancePlanDetailStateEnum.TO_BE_COLLECTED.getValue()) // 当前为待收款
                .lt(OrderBillEntity::getPayEndTime, DateUtil.formatDate(new Date())) // 已超出付款期限
        );
        log.info("查询到需要更新的账单 {} 条", list.size());
        if (ToolUtil.isEmpty(list)) {
            return false;
        }
        // 将账单设置为已逾期
        list.forEach(x -> x.setPayState(FinanceEnums.FinancePlanDetailStateEnum.OVERDUE.getValue()));
        boolean flag = this.updateBatchById(list, list.size());
        log.info("批量更新账单状态为已逾期：{}", flag);
        if (!flag) {
            log.info("批量更新账单付款状态失败！");
            return false;
        }
        // 更新收款计划状态
        Set<Long> planPayIds = list.stream().map(OrderBillEntity::getPlanPayId).collect(Collectors.toSet());
        financePlanPayService.checkPlanPayUpdateState(planPayIds);
        return true;
    }

    /**
     * 计算并修改状态
     *
     * @param old
     * @param isPayStateUpdate     更新账单状态，false  只计算， true 计算并更新数据库状态
     * @param isPlanPayStateUpdate 更新计划状态，false  只计算， true 计算并更新数据库状态
     * @return
     */
    public boolean calculationPayState(OrderBillEntity old, boolean isPayStateUpdate, boolean isPlanPayStateUpdate) {
        BigDecimal actualAmount = old.getActualAmount() != null ? old.getActualAmount() : BigDecimal.ZERO;
        BigDecimal amountActuallyReceived = old.getAmountActuallyReceived() != null ? old.getAmountActuallyReceived() : BigDecimal.ZERO;
        if (ToolUtil.isEmpty(actualAmount)) {
            actualAmount = new BigDecimal("0");
        }
        Long oldBillId = old.getId();
        if (amountActuallyReceived.compareTo(actualAmount) >= 1) {
            old.setPayState(FinanceEnums.FinancePlanDetailStateEnum.COLLECTED.getValue());
        } else if (amountActuallyReceived.compareTo(actualAmount) == 0) {
            old.setPayState(FinanceEnums.FinancePlanDetailStateEnum.RECEIVED.getValue());
        } else {
            old.setPayState(FinanceEnums.FinancePlanDetailStateEnum.TO_BE_COLLECTED.getValue());
        }

        if (FinanceEnums.FinancePlanDetailStateEnum.OVERDUE.getValue().equals(old.getPayState())
                || FinanceEnums.FinancePlanDetailStateEnum.TO_BE_COLLECTED.getValue().equals(old.getPayState())) {
            if (ToolUtil.isNotEmpty(old.getPayEndTime())) {
                Date currentDate = DateUtil.parseDate(DateUtil.formatDate(new Date()));
                Date payEndTime = DateUtil.parseDate(DateUtil.formatDate(old.getPayEndTime()));
                if (currentDate.getTime() > payEndTime.getTime()) {
                    old.setPayState(FinanceEnums.FinancePlanDetailStateEnum.OVERDUE.getValue());
                }
            }
        }

        boolean flag = false;
        if (isPayStateUpdate) {
            flag = this.updateById(old);
        }
        if (isPlanPayStateUpdate) {
            flag = financePlanPayService.checkPlanPayUpdateState(old.getPlanPayId());
        }
        return flag;
    }

    // 检查账单能否删除，返回不能删除的账单id及原因
    public List<AuditResult> checkBillDel(Collection<Long> billIds) {
        List<AuditResult> resultList = new ArrayList<>();
        billIds = billIds.stream().distinct().collect(Collectors.toList());
        // 判断账单是否有收款
        List<Long> collectBillIds = checkHasCollect(billIds);
        if (ToolUtil.isNotEmpty(collectBillIds)) {
            for (Long billId : collectBillIds) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("账单存在收款记录");
                resultList.add(resp);
            }
        }
        // 判断账单是否有正在支付的订单
        List<Long> payingBillIds = payOrderService.getPayingBillIds(billIds);
        if (ToolUtil.isNotEmpty(payingBillIds)) {
            for (Long billId : payingBillIds) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("账单正在支付");
                resultList.add(resp);
            }
        }
        // 判断账单是否有减免
        for (Long billId : billIds) {
            if (existsExemption(billId)) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("账单存在减免记录");
                resultList.add(resp);
            }
        }
        // 判断账单是否已开票
        List<Long> receiptBillIds = checkHasReceipt(billIds);
        if (ToolUtil.isNotEmpty(receiptBillIds)) {
            for (Long billId : receiptBillIds) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("账单已开具票据");
                resultList.add(resp);
            }
        }
        // 判断账单是否有坏账申请
        List<Long> badDebts = checkBadDebts(billIds);
        if (ToolUtil.isNotEmpty(badDebts)) {
            for (Long billId : badDebts) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("账单存在坏账记录");
                resultList.add(resp);
            }
        }
        return resultList;
    }

    // 检查账单是否有审核中的减免、坏账、退款
    public List<AuditResult> checkBillAudit(Collection<Long> billIds) {
        if (ToolUtil.isEmpty(billIds)) {
            return new ArrayList<>();
        }
        List<AuditResult> resultList = new ArrayList<>();
        // 判断账单是否存在申请中的减免记录
        for (Long billId : billIds) {
            if (existsAuditExemption(billId)) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("存在申请中的减免记录");
                resultList.add(resp);
            }
        }
        // 判断账单是否有申请中的坏账
        List<Long> badDebts = checkAuditBadDebts(billIds);
        if (ToolUtil.isNotEmpty(badDebts)) {
            for (Long billId : badDebts) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("存在申请中的坏账记录");
                resultList.add(resp);
            }
        }
        // 判断账单是否有申请中的退款
        List<Long> refundBillIds = checkAuditRefund(billIds);
        if (ToolUtil.isNotEmpty(refundBillIds)) {
            for (Long billId : refundBillIds) {
                AuditResult resp = new AuditResult();
                resp.setBillId(billId);
                resp.setResult("存在申请中的退款记录");
                resultList.add(resp);
            }
        }
        return resultList;
    }

    public PageListVO<OrderBillGroupResResp> pageGroupByResource(OrderBillGroupResQuery query) {
        Page<OrderBillGroupResResp> page = new Page<>(query.getCurrentPage(), query.getPageSize());
        if (ToolUtil.isNotEmpty(query.getFilter())) {
            List<Long> resIds = getPmsResourcesIdsByFilter(query.getOrganizeId(), query.getFilter());
            List<Long> carIds = this.getPmsCarportIdsByFilter(query.getOrganizeId(), query.getFilter());
            query.setFilterResIds(resIds);

        }
        IPage<OrderBillGroupResResp> result = this.getBaseMapper().listGroupByResource(page, query,
                commonService.conditionSql(ModuleConstant.RESOURCE_BILL, query.getOrganizeId()));
        if (ToolUtil.isNotEmpty(result.getRecords())) {
            Set<Long> resIds = result.getRecords().stream().filter(x -> ObjectUtil.equal(x.getResourceMode(), FinanceEnums.ResourceModeEnum.RESOURCE.getValue())).map(OrderBillGroupResResp::getResourceId).collect(Collectors.toSet());
            Set<Long> carIds = result.getRecords().stream().filter(x -> ObjectUtil.equal(x.getResourceMode(), FinanceEnums.ResourceModeEnum.CARPORT.getValue())).map(OrderBillGroupResResp::getResourceId).collect(Collectors.toSet());
            List<PmsResourcesByIdsResp> resListByIds = getResListByIds(resIds);
            List<PmsCarportResp> carportListByIds = getPmsCarportListByIds(carIds);
            List<String> orgIds = result.getRecords().stream().filter(x -> ToolUtil.isNotEmpty(x.getOrganizeId()))
                    .map(OrderBillGroupResResp::getOrganizeId).distinct().collect(Collectors.toList());
            Map<String, String> orgMap = organizeApi.getOrgMap(orgIds);
            result.getRecords().forEach(z -> {
                String no = "";
                if (ObjectUtil.equal(z.getResourceMode(), FinanceEnums.ResourceModeEnum.RESOURCE.getValue())) {
                    PmsResourcesByIdsResp resourceResp = StreamHelper.findListFirst(resListByIds, c -> ObjectUtil.equal(ToolUtil.isEmpty(z.getResourceId()) ? "" : z.getResourceId().toString(), c.getId()), new PmsResourcesByIdsResp());
                    no = resourceResp.getResourcesNo();
                } else if (ObjectUtil.equal(z.getResourceMode(), FinanceEnums.ResourceModeEnum.CARPORT.getValue())) {
                    PmsCarportResp resourceResp = StreamHelper.findListFirst(carportListByIds, x -> ObjectUtil.equal(z.getResourceId(), x.getId()), new PmsCarportResp());
                    no = resourceResp.getResourceNo();
                }
                if (ToolUtil.isNotEmpty(z.getOrganizeId())) {
                    z.setOrganizeName(orgMap.get(z.getOrganizeId()));
                }
                z.setResourceNo(no);
            });
        }
        return new PageListVO<>(result.getRecords(), result.getCurrent(), result.getSize(), result.getTotal());
    }

    public void refreshActualAmount(List<OrderBillEntity> list) {
        if (ToolUtil.isEmpty(list)) {
            return;
        }
        log.info("refreshActualAmount  ====> list = {}", list.size());

        Set<Long> billIds = list.stream().map(OrderBillEntity::getId).collect(Collectors.toSet());
        Map<Long, List<BillReceivableDetailEntity>> detailMap = billReceivableDetailService.getDetailMap(billIds);
        for (OrderBillEntity orderBill : list) {
            List<BillReceivableDetailEntity> detailList = detailMap.get(orderBill.getId());

        if (updateList.size() > 0) {
            log.info("refreshActualAmount  ====> updateList = {}", updateList.size());
            this.updateBatchById(updateList, updateList.size());
        }
    }

    public void setPublishStatus(Collection<OrderBillEntity> list, boolean save) {
        if (ToolUtil.isEmpty(list)) {
            return;
        }
        // 获取自动发布配置
        OrderBillEntity firstBill = list.iterator().next();
        BusinessAutoConfigQuery autoConfigQuery = new BusinessAutoConfigQuery();
        autoConfigQuery.setOrganizeId(firstBill.getOrganizeId());
        autoConfigQuery.setAutoType(FinanceEnums.BusinessAutoConfigAutoTypeEnum.BILL_PUBLISH.getValue());
        autoConfigQuery.setBusinessType(FinanceEnums.BusinessAutoConfigBusinessTypeEnum.PUBLISH_BILL.getValue());
        autoConfigQuery.setStatus(FinanceEnums.BusinessAutoConfigStatusEnum.ENABLE.getValue());
        List<BusinessAutoConfigResp> configList = businessAutoConfigService.getBusinessAutoConfigList(autoConfigQuery);
        boolean matchAllExpenseFlag = false;
        Set<Long> matchExpenseIds = new HashSet<>();
        if (ToolUtil.isNotEmpty(configList)) {
            matchAllExpenseFlag = configList.stream().anyMatch(x -> ObjectUtil.equal(x.getExpenseApplyType(), FinanceEnums.BusinessAutoConfigExpenseApplyTypeEnum.ALL.getValue()));
            if (!matchAllExpenseFlag) {
                for (BusinessAutoConfigResp config : configList) {
                    if (ObjectUtil.equal(config.getExpenseApplyType(), FinanceEnums.BusinessAutoConfigExpenseApplyTypeEnum.PART.getValue())
                            && ToolUtil.isNotEmpty(config.getExpenseIds())) {
                        matchExpenseIds.addAll(RespTools.strToLongList(config.getExpenseIds()));
                    }
                }
            }
        }

        // 设置发布状态
        UserInfo userInfo = userProvider.get();
        for (OrderBillEntity orderBill : list) {
            if (matchAllExpenseFlag || matchExpenseIds.contains(orderBill.getExpenseId())) {
                if (orderBill.getPublishStatus() == null) {
                    orderBill.setPublishStatus(FinanceEnums.OrderBillPublishStatusEnum.YES.getValue());
                    orderBill.setPublishUserId(userInfo.getUserId());
                    orderBill.setPublishUserName(userInfo.getUserName());
                    orderBill.setPublishTime(new Date());
                }
            }
        }
        if (save) {
            this.saveOrUpdateBatch(list, list.size());
        }
    }

    /**
     * 更新账单减免、坏账、调整金额
     */
    public void updateCorrectionAmount(Collection<OrderBillEntity> billList, boolean isUpdate) {
        if (ToolUtil.isEmpty(billList)) {
            return;
        }
        // 获取账单的减免、坏账、调整金额
        String organizeId = billList.iterator().next().getOrganizeId();
        Set<Long> billIds = billList.stream().map(OrderBillEntity::getId).collect(Collectors.toSet());
        List<OrderBillResp> sumRespList = this.getBaseMapper().sumCorrectionAmount(organizeId, billIds);
        Map<Long, OrderBillResp> sumRespMap = new HashMap<>();
        if (ToolUtil.isNotEmpty(sumRespList)) {
            Map<Long, OrderBillResp> map = sumRespList.stream().collect(Collectors.toMap(OrderBillResp::getBillId, a -> a, (k1, k2) -> k1));
            sumRespMap.putAll(map);
        }
        // 赋值
        for (OrderBillEntity bill : billList) {
            OrderBillResp sumResp = sumRespMap.get(bill.getId());
            if (sumResp != null) {
                bill.setOriginalReceivableAmount(sumResp.getOriginalReceivableAmount());
                bill.setExemptionAmount(sumResp.getExemptionAmount());
                bill.setBadDebtsAmount(sumResp.getBadDebtsAmount());
                bill.setCorrectAmount(sumResp.getCorrectAmount());
            } else {
                bill.setOriginalReceivableAmount(BigDecimal.ZERO);
                bill.setExemptionAmount(BigDecimal.ZERO);
                bill.setBadDebtsAmount(BigDecimal.ZERO);
                bill.setCorrectAmount(BigDecimal.ZERO);
            }
        }
        if (isUpdate) {
            this.saveOrUpdateBatch(billList, billList.size());
        }
    }

    /**
     * 更新所有组织账单减免、坏账、调整金额
     */
    public Boolean updateOrgCorrectionAmount(String organizeId) {
        // 获取组织
        List<String> organizeIds = null;
        if (ToolUtil.isNotEmpty(organizeId)) {
            organizeIds = organizeApi.treeIdsByOrganizeId(organizeId);
        } else {
            ActionResult<List<OrganizeEntity>> actionResult = organizeApi.getAllOrganizeList();
            if (actionResult != null && ToolUtil.isNotEmpty(actionResult.getData())) {
                organizeIds = actionResult.getData().stream().map(OrganizeEntity::getId).distinct().collect(Collectors.toList());
            }
        }
        if (ToolUtil.isEmpty(organizeIds)) {
            log.info("找不到组织，终止更新");
            return false;
        }
        // 遍历更新
        for (String orgId : organizeIds) {
            BatchLoadUtils.loadByPage((currPage, pageSize) -> {
                Page<OrderBillEntity> result = this.page(new Page<>(currPage, pageSize), Wrappers.<OrderBillEntity>lambdaQuery()
                        .eq(OrderBillEntity::getOrganizeId, orgId)
                        .eq(OrderBillEntity::getChangeStatus, FinanceEnums.ChangeStatusEnum.EFFECT.getValue())
                );
                if (result == null || ToolUtil.isEmpty(result.getRecords())) {
                    return null;
                }
                List<OrderBillEntity> list = result.getRecords();
                updateCorrectionAmount(list, true); // 执行更新
                return list;
            });
        }
        return true;
    }

    @Async
    public void writeExcelOrderBillZipData(OrderBillQuery query, UserInfo userInfo) {
        Integer threshold = 1000;
        // 获取数据进行分割
        if (ToolUtil.isNotEmpty(query.getThreshold())) {
            threshold = query.getThreshold();
        }
        List<OrderBillResp> respList = getOrderBillList(new Page(1, -1L), query);
        List<List<OrderBillResp>> partition = Lists.partition(respList, threshold);
        int count = respList.size();
        int pageSize = threshold;
        int tableNum = partition.size();
        String randomUUID = IdUtil.randomUUID();
        // 将数据多线程方式导出到excel
        CountDownLatch latch = new CountDownLatch(tableNum);
        String saveFileName = String.format("%s%s%s.zip", jnpf.util.DateUtil.dateNow(DatePattern.PURE_DATE_PATTERN), Enums.FileBusinessTypeEnum.FINANCE_ORDER_BILL.toString(), randomUUID);
        ExcelExportZip excelExportZip = ExcelExportZip.build(OrderBillExcelResp.class).user(userInfo).message("financial", "billDetail").count(count).businessTypeEnum(Enums.FileBusinessTypeEnum.FINANCE_ORDER_BILL).zipFileName(saveFileName);
        FileImportExportBatchEntity fileImportExportBatchEntity = excelExportZip.startExportBatch();
        try {
            for (int i = 0; i < tableNum; i++) {
                int finalI = i;
                    exportDataToExcel(latch, pageSize, finalI + 1, query, randomUUID, partition.get(finalI));
                log.info(Enums.FileBusinessTypeEnum.FINANCE_ORDER_BILL + "-" + i + "文件结束");
            }
            latch.await(RedisUtil.CAHCETIME * 5, TimeUnit.SECONDS);
            excelExportZip.startZip(saveFileName, randomUUID, fileImportExportBatchEntity);
        } catch (InterruptedException e) {
            excelExportZip.exceptionExportBatch(fileImportExportBatchEntity);
        }
    }

    private void exportDataToExcel(CountDownLatch latch, int pageSize, int i, OrderBillQuery query, String fileUUID, List<OrderBillResp> dataList) {
        List<OrderBillResp> respList = new ArrayList<>(dataList);
        List<String> orgIds = respList.stream().filter(x -> ToolUtil.isNotEmpty(x.getOrganizeId()))
                .map(OrderBillResp::getOrganizeId).distinct().collect(Collectors.toList());
        Map<String, String> orgMap = organizeApi.getOrgMap(orgIds);
        List<OrderBillExcelResp> list = new ArrayList<>();
        for (OrderBillResp x : respList) {
            OrderBillExcelResp item = new OrderBillExcelResp();
            item.setProductKey(x.getProductKey());
            item.setResourceName(x.getResourceName());
            item.setExpenseName(x.getExpenseName());
            if (x.getLeaseStartTime() != null && x.getLeaseEndTime() != null) {
                String s = String.format("%s-%s", DateUtil.formatDate(x.getLeaseStartTime()), DateUtil.formatDate(x.getLeaseEndTime()));

    public void exportBillGroupByCustomerZipV2(OrderBillGroupQuery query, UserInfo userInfo) {

        // -------------1. 基础数据准备---------------

        Enums.FileBusinessTypeEnum businessTypeEnum = Enums.FileBusinessTypeEnum.FINANCE_BILL_DETAIL_CUSTOMER_LIST;
        String globalPath = configValueUtil.getWebAnnexFilePath();

        int eachExcelCount = 1000; // 1w
        int eachPageCount = 100; // 1k

        String randomUUID = IdUtil.randomUUID();
        String fileDirPath = globalPath + "/zip" + randomUUID; // 文件加的路径
        File fileZip = new File(fileDirPath);
        if (!fileZip.exists()) {
            boolean mkdirs = fileZip.mkdirs();
        }

        String zipExcelFileName = String.format(
                "%s%s%s.zip", DateUtil.format(new Date(),DatePattern.PURE_DATE_PATTERN), businessTypeEnum, randomUUID
        );
        String zipExcelPath = String.format(
                "%s%s", globalPath + "/", zipExcelFileName
        );

        // -------------2. 获取数据totalCount---------------
        OrderBillMapper orderBillMapper = this.getBaseMapper();
        List<Long> connectIdList = orderBillMapper.listGroupConnectIdByCustomer(
                query,
                commonService.conditionSql(ModuleConstant.CUSTOMER_BILL, query.getOrganizeId())
        );

        OrderBillQuery billQuery = new OrderBillQuery();
        billQuery.setOrganizeId(query.getOrganizeId());
        billQuery.setConnectIds(connectIdList);
        formatQuery(billQuery);
        int totalDataCount = orderBillMapper.getTotalCountByQuery(billQuery, conditionSql());


        // -------------3. 参数绑定---------------
        int totalPage = totalDataCount / eachPageCount + (totalDataCount % eachPageCount == 0 ? 0 : 1); // 查询分页数量
        int totalExcelFile = totalDataCount / eachExcelCount + (totalDataCount % eachExcelCount == 0 ? 0 : 1); // excel文件数量

        List<Integer> pageNumList = new ArrayList<>(totalPage);
        for (int i = 1; i < totalPage + 1; i++) {
            pageNumList.add(i);
        }
        // 每个excel 处理的
        List<List<Integer>> partition = Lists.partition(pageNumList, pageNumList.size() /  totalExcelFile + (pageNumList.size() %  totalExcelFile == 0 ? 0 : 1));
        HashMap<Integer, List<Integer>> excelFileBindPagesMap = new HashMap<>();
        for (int i = 0; i < totalExcelFile; i++) {
            List<Integer> pageNums = partition.get(i);
            excelFileBindPagesMap.put(i + 1,pageNums);
        }

        // -------------4. 创建导入记录---------------
        FileImportExportBatchEntity fileBatch = new FileImportExportBatchEntity();
        String downloadFilePath = UploaderUtil.uploaderFile(userInfo.getUserId() + "#" + zipExcelFileName + "#" + FileTypeEnum.ANNEX);
        fileBatch.setOrganizeId(query.getOrganizeId());
        JSONObject object = new JSONObject();
        object.put("organizeId", query.getOrganizeId());
        fileBatch.setReqparam(JSON.toJSONString(object));
        fileBatch.setType(Enums.FileImportExportTypeEnum.EXPORT.getValue());
        fileBatch.setBusinessType(businessTypeEnum.getValue());
        fileBatch.setOperatingUserName(userInfo.getUserName());
        fileBatch.setState(Enums.FileImportExportStateEnum.HAVE_IN_HAND.getValue());
        fileBatch.setTotalData(totalDataCount);
        fileBatch.setFilePath(downloadFilePath);
        fileImportExportBatchService.save(fileBatch);

        // -------------4. 分页查询处理每个excel 文件---------------
        List<String> filePathList = new ArrayList<>();
        OraganizeInfoVO oraganizeInfoVO = organizeApi.getById(query.getOrganizeId());

        for (Map.Entry<Integer, List<Integer>> entry : excelFileBindPagesMap.entrySet()) {
            // 处理每个文件的逻辑
            Integer key = entry.getKey(); // excel 文件序号
            List<Integer> pages = entry.getValue(); // 每个excel文件分配的分页list
            // 文件名字
            String filePath = String.format(
                    "%s%s%s%s%s.xls",fileDirPath + "/", DateUtil.format(new Date(),DatePattern.PURE_DATE_PATTERN), businessTypeEnum, "-" + key + "-", RandomUtil.uuId()
            );
            filePathList.add(filePath);

            // easy-excel build
            ExcelWriter excelWriter = EasyExcel.write(filePath, OrderBillDetailExcelResp.class).build();
            WriteSheet writeSheet = EasyExcel.writerSheet(businessTypeEnum + "数据导出").build();

            // 处理每个excel 文件中的分页列表
            for (Integer page : pages) {
                // 开始执行查询
                Page<OrderBillEntity> entityPage = new Page<>(page,eachPageCount);
                List<OrderBillResp> orderBillList = this.getOrderBillList(entityPage,billQuery);

                if (ToolUtil.isEmpty(orderBillList)) {
                    throw new ServiceException(FinanceExceptionEnum.DATA_EMPTY);
                }
                // 对象转化
                List<OrderBillResp> billRespList = orderBillList.stream().map(x -> {
                    x.setOrganizeName(oraganizeInfoVO.getFullName());
                    return BeanUtil.toBean(x, OrderBillResp.class);
                }).collect(Collectors.toList());
                // 填充VO对象
                List<OrderBillDetailExcelResp> excelDataList = orderBillImportExportService.writeDetailExcelDataList(billRespList, query.getOrganizeId(), query, Enums.FileBusinessTypeEnum.FINANCE_BILL_DETAIL_CUSTOMER_LIST);
                // 查询合并表头对象
                List<ExcelMergedRegion> mergedRegions = orderBillImportExportService.writeDetailExcelDataListGroupMap(excelDataList, query.getOrganizeId(), query, Enums.FileBusinessTypeEnum.FINANCE_BILL_DETAIL_CUSTOMER_LIST);

                // 开始写 excel 数据 [core]
                excelWriter.write(excelDataList, writeSheet);
            }
            excelWriter.close();
        }
        //-------------5. 合并zip文件---------------
        File excelZipFile = ZipUtil.zip(fileDirPath,  zipExcelPath);

        // -------------6. 解压完成---------------
        if (excelZipFile.exists()) {
            // 对象流的转化
            DiskFileItemFactory diskFileItemFactory = new DiskFileItemFactory();
            FileItem item = diskFileItemFactory.createItem(
                    "file",
                    MediaType.MULTIPART_FORM_DATA_VALUE,
                    true,
                    excelZipFile.getName()
            );
            try (InputStream input = Files.newInputStream(excelZipFile.toPath());
                 OutputStream outputStream = item.getOutputStream()
            ) {
                 IOUtils.copy(input, outputStream);
            } catch (Exception e) {
                throw new IllegalArgumentException("Invalid file: " + e, e);
            }
            CommonsMultipartFile multipartFile = new CommonsMultipartFile(item);
            // 上传oss 文件-文件路径-文件类型-文件名字
            FileUtil.upFile(multipartFile, configValueUtil.getWebAnnexFilePath(), FileTypeEnum.ANNEX, zipExcelFileName);
        }

        // -------------7. 更新导入记录---------------
        fileBatch.setState(Enums.FileImportExportStateEnum.COMPLETE.getValue());
        fileBatch.setSuccessData(totalDataCount);
        fileImportExportBatchService.updateById(fileBatch);

        // -------------8. 发送消息---------------

        String project = "financial";
        String module = "customerBill";
        String exportType = "2";

        SimpleDateFormat yyMMdd = new SimpleDateFormat("yyMMdd");
        String firstToSixthNumber = yyMMdd.format(new Date());
        String messageText=firstToSixthNumber+businessTypeEnum+"("+generateSerialNumber(userInfo,businessTypeEnum)+").zip";
        SentMessageModel model = new SentMessageModel();
        List<MessageUserModel> userList = new ArrayList<>();
        MessageUserModel user = new MessageUserModel();
        user.setUserId(userInfo.getUserId());
        user.setOrganizeId(userInfo.getOrganizeId());
        userList.add(user);
        Map<String, Object> message = new HashMap<>(16);
        message.put("project",project);
        message.put("module", module);
        if(ToolUtil.isNotEmpty(exportType)){
            message.put("exportType",exportType);
        }
        message.put("type","5");
        message.put("message",messageText);
        String bodyText = JsonUtil.getObjectToString(message);

        model.setBodyText(bodyText);
        model.setTitle(messageText);
        model.setUserList(userList);
        model.setOrganizeId(userInfo.getOrganizeId());
        model.setBusinessType(1);
        //socket
        PushMessageModel pushMessageModel=new PushMessageModel();
        pushMessageModel.setMethod("exportExcel");
        pushMessageModel.setToUserIds(userList.stream().map(MessageUserModel::getUserId).collect(Collectors.toList()));
        pushMessageModel.setBodyText(bodyText);
        noticeApi.sentMessage(model);
        noticeApi.pushMessage(pushMessageModel);

        new File(fileDirPath).deleteOnExit(); // 删除生成的文件加
    }

    public  String generateSerialNumber(UserInfo userInfo,Enums.FileBusinessTypeEnum businessTypeEnum) {

        SimpleDateFormat yyMMdd = new SimpleDateFormat("yyMMdd");
        String firstToSixthNumber = yyMMdd.format(new Date());
        String key=userInfo.getUserId()+firstToSixthNumber+businessTypeEnum.getValue().toString();

        }else{
            int string =Integer.parseInt(redisUtil.getString(key).toString().replace("\"", ""));
            redisUtil.insert(key,string+1,RedisUtil.CAHCEDAY);
        }

        return String.valueOf(redisUtil.getString(key));
    }

            throw new ServiceException(FinanceExceptionEnum.DATA_EMPTY);
        }
        List<Long> redIds = list.stream().map(OrderBillGroupResResp::getResourceId).collect(Collectors.toList());
        OrderBillQuery billQuery = new OrderBillQuery();
        billQuery.setOrganizeId(query.getOrganizeId());
        billQuery.setResourceIds(redIds);
        billQuery.setPageSize(-1L);
        billQuery.setCurrentPage(1L);
        List<OrderBillResp> orderBillList = this.getOrderBillList(billQuery);
        List<OrderBillResp> billRespList = orderBillList.stream().map(x -> BeanUtil.toBean(x, OrderBillResp.class)).collect(Collectors.toList());
        List<OrderBillDetailExcelResp> orderBillDetailExcelResps = orderBillImportExportService.writeDetailExcelDataList(billRespList, query.getOrganizeId(), query, Enums.FileBusinessTypeEnum.FINANCE_BILL_DETAIL_RES_LIST);
        List<List<OrderBillDetailExcelResp>> partition = Lists.partition(orderBillDetailExcelResps, threshold);
        // 将数据多线程方式导出到excel
        CountDownLatch latch = new CountDownLatch(partition.size());
        String saveFileName = String.format("%s%s%s.zip", jnpf.util.DateUtil.dateNow(DatePattern.PURE_DATE_PATTERN), Enums.FileBusinessTypeEnum.FINANCE_BILL_DETAIL_RES_LIST.toString(), randomUUID);
        ExcelExportZip excelExportZip = ExcelExportZip.build(OrderBillDetailExcelResp.class).user(userInfo).oId(query.getOrganizeId()).exportType("2").message("financial", "resourceBill").count(billRespList.size()).businessTypeEnum(Enums.FileBusinessTypeEnum.FINANCE_BILL_DETAIL_RES_LIST).zipFileName(saveFileName);
        FileImportExportBatchEntity fileImportExportBatchEntity = excelExportZip.startExportBatch();
        for (int i = 0; i < partition.size(); i++) {
            int finalI = i;
            TimeInterval timer = DateUtil.timer();
            exportDataToExcelDetailBill(latch, pageSize, finalI + 1, query, randomUUID, partition.get(finalI));
            log.info(Enums.FileBusinessTypeEnum.FINANCE_BILL_DETAIL_RES_LIST + "-" + finalI + "花费时间:" + timer.interval());
        }
        try {
            latch.await(RedisUtil.CAHCETIME * 5, TimeUnit.SECONDS);
            excelExportZip.startZip(saveFileName, randomUUID, fileImportExportBatchEntity);
        } catch (InterruptedException e) {
            excelExportZip.exceptionExportBatch(fileImportExportBatchEntity);
        }
    }

    public Boolean fillMeterFullName() {
        BatchLoadUtils.loadByPage((currentPage, pageSize) -> {
            IPage<OrderBillPropertyResp> page = this.getBaseMapper().listFillMeterFullNameBillList(new Page<>(currentPage, pageSize));
            if (page == null || ToolUtil.isEmpty(page.getRecords())) {
                return null;
            }
            List<OrderBillPropertyResp> list = page.getRecords();
            Map<Long, OrderBillPropertyResp> respMap = list.stream().collect(Collectors.toMap(OrderBillPropertyResp::getBillId, a -> a, (k1, k2) -> k1));
            List<OrderBillEntity> orderBillList = this.listByIds(respMap.keySet());
            for (OrderBillEntity bill : orderBillList) {
                try {
                    OrderBillPropertyResp resp = respMap.get(bill.getId());
                    if (resp != null) {
                        List<TaskDetailItemBO> itemList = JSONArray.parseArray(resp.getDetailJson(), TaskDetailItemBO.class);
                        String meterFullName = itemList.stream()
                                .map(x -> UtilHelps.spConcatKey("/", x.getSerialNumber(), x.getMeterName()))
                                .collect(Collectors.joining(","));
                        bill.setMeterFullName(meterFullName);
                    }
                } catch (Exception e) {
                    log.error("OrderBillService -> 补全meterFullName(旧数据处理)解析异常", e);
                }
            }
            this.updateBatchById(orderBillList);
            return list;
        });
        return true;
    }
