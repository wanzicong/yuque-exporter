## nginx 安装 
```shell
前提条件:
-bash: wget: command not found
# 安装 wget 
sudo yum install wget

#1、下载nginx依赖 【进入指定目录进行安装 /root】
wget http://nginx.org/packages/centos/7/noarch/RPMS/nginx-release-centos-7-0.el7.ngx.noarch.rpm

#2、运行nginx依赖包
rpm -ivh nginx-release-centos-7-0.el7.ngx.noarch.rpm

#3、安装nginx
sudo yum install nginx –y

#4、配置nginx
systemctl start nginx
# 查看 启动状态
systemctl status nginx.service
systemctl enable nginx


===============================================================================================
#5、安装logrotate分隔日志
yum -y install logrotate

#6、写入分割脚本
cat > /etc/logrotate.d/nginx << EOF
/var/log/nginx/*.log /usr/local/nginx/logs/*.log {
    create 0664 root root
    daily
    rotate 30
    size +100M
    missingok
    notifempty
    compress
    sharedscripts
    postrotate
        /bin/kill -USR1 \`cat /usr/local/nginx/logs/nginx.pid 2>/dev/null\` 2>/dev/null || true
    endscript
}
EOF

#create 0664 root root # 使用该模式创建日志文件
#daily # 日志轮询周期，weekly,monthly,yearly
#rotate 30 # 保存30天数据，超过的则删除
#size +100M  # 超过100M时分割，单位K,M,G，优先级高于daily
#missingok  # 如果没有日志文件也不报错
#notifempty # 日志为空时不进行切换，默认为ifempty
#compress  # 切割后压缩，也可以为nocompress
#sharedscripts # 所有的文件切割之后只执行一次下面脚本

#7、验证 手动强制切割日志
logrotate --force /etc/logrotate.d/nginx

#8、创建ssl目录存放ssl证书
mkdir /etc/nginx/ssl

nginx 目录
/usr/share/nginx/html
```
