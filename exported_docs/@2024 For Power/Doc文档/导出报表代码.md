    private List<ReportCollectionRateResp> dealResp(List<ReportCollectionRateResp> list) {
        if (ToolUtil.isEmpty(list)) {
            return list;
        }
        List<ReportCollectionRateResp> newList = new ArrayList<>();
        Long projectId = list.get(0).getProjectId();
        for (int i = 0; i < list.size(); i++) {
            ReportCollectionRateResp resp = list.get(i);
            calcResp(resp);
            if (i == list.size() - 1) {
                if (!projectId.equals(resp.getProjectId())) {
                    ReportCollectionRateResp summary = getSummary(resp.getOrganizeId(), projectId, resp.getBelongTime());
                    newList.add(summary);
                }
                newList.add(resp);
                ReportCollectionRateResp summary = getSummary(resp.getOrganizeId(), resp.getProjectId(), resp.getBelongTime());
                newList.add(summary);
                break;
            }
            if (projectId.equals(resp.getProjectId())) {
                newList.add(resp);
            } else {
                ReportCollectionRateResp summary = getSummary(resp.getOrganizeId(), projectId, resp.getBelongTime());
                newList.add(summary);
                newList.add(resp);
                projectId = resp.getProjectId();
            }
        }
        return newList;
    }

    public void calcRate(ReportCollectionRateResp resp) {
        if (resp == null) {
            return;
        }
        // 清欠率
        resp.setLiquidationRate(getRate(resp.getPastActualTotal(), resp.getPastDebt()));
        // 本期收缴率
        resp.setCurPeriodRate(getRate(resp.getActualTotal(), resp.getCurPeriodIncome()));
        // 收缴进度
        resp.setYearRate(getRate(resp.getYearReceivedTotal(), resp.getYearReceivableTotal()));
        resp.setYearFullRate(resp.getYearRate());
    }

    public BigDecimal getRate(BigDecimal divisor, BigDecimal dividend) {
        if (divisor == null || divisor.compareTo(BigDecimal.ZERO) <= 0
                || dividend == null || dividend.compareTo(BigDecimal.ZERO) <= 0) {
            return BigDecimal.ZERO;
        }
        BigDecimal rate = NumberUtil.mul(NumberUtil.div(divisor, dividend), new BigDecimal("100"));
        return UtilHelps.convertBigDecimal(0, rate);
    }
